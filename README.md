# üöÄ Getting Started
## Quick Start
bash# Clone the repository
git clone https://github.com/apollo-platform/apollo.git
cd apollo

# Install dependencies
./scripts/setup/install.sh

# Configure environment
cp .env.example .env.local
# Edit .env.local with your settings

# Initialize databases
./scripts/setup/init-database.sh

# Set up AI models
./scripts/setup/setup-ai-models.sh

# Start the platform
docker-compose up -d

# Verify installation
./scripts/setup/verify-installation.sh
Voice Recognition Setup
bash# Install audio processing dependencies
pip install openai-whisper speechbrain resemblyzer

# Configure voice recognition
./scripts/setup/setup-voice-recognition.sh

# Load Ignatova voice profile
python ai-engine/audio-intelligence/voice-databases/load-ignatova-profile.py
Facial Recognition Setup
bash# Install facial recognition dependencies
pip install face-recognition opencv-python

# Configure facial recognition
./scripts/setup/setup-facial-recognition.sh

# Load FBI Most Wanted profiles
python services/facial-recognition/load-fbi-profiles.py
Blockchain Forensics Setup
bash# Configure blockchain analysis
./scripts/setup/setup-blockchain-forensics.sh

# Load OneCoin specific data
python services/blockchain-forensics/load-onecoin-data.py
